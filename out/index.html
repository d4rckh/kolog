Kolog

Advanced logging util for nodejs

Installing:

NPM:

npm i kolog

Yarn:

yarn add kolog

Example:

var Kolog = require('../src/index.js')

var logger = new Kolog.Logger({scope: 'normal logs'})

console.log('--LOGGER--')

logger.log('genericLog')
logger.warn('Warning')
logger.error('error')
logger.success('success')
logger.info('info')

Docs: https://node-kolog.github.io
